class PcProduct extends MY_Pc_Controller {
   public function __construct() {
       parent::__construct();

       $this->load->model('pc/Pc_product_model', 'product');
       $this->load->model('pc/Pc_category_model', 'pc_category');
   }

   /**
    * 列表
    */
   public function index() {
       //当前第几页
       $cur_page = intval($this->input->get('per_page'));
       $cur_page = $cur_page < 1 ? 1 : $cur_page;

       //每页的显示条数
       $page_size = 10;
       $where     = ' 1=1';
       $where_str = '_iframe=1';

       if (isset($_GET['status']) && !empty($_GET['status'])) {
           $status    = intval(trim($_GET['status'])) - 1;
           $where     .= ' AND STATUS = ' . $status;
           $where_str .= '&status=' . $_GET['status'];
       }
       if (isset($_GET['cate_id']) && !empty($_GET['cate_id'])) {
           $cate_id   = $this->_replaceParamVal($_GET['cate_id']);
           $where     .= ' AND CATEGORY_ID = \'' . $cate_id . '\'';
           $where_str .= '&cate_id=' . $_GET['cate_id'];
       }
       if (isset($_GET['app_search_input']) && !empty($_GET['app_search_input'])) {
           $where     .= ' AND NAME LIKE \'%' . $this->_replaceParamVal($_GET['app_search_input']) . '%\'';
           $where_str .= '&app_search_input=' . $_GET['app_search_input'];
       }

       //图片链接
       $pic_url = $GLOBALS['CHNAPP_UPLOAD_URL'];
       $this->assign('pic_url', $pic_url);

       $count = $this->product->get_count($where);

       $base_url = uri_string();
       if (strpos(uri_string(), '?') !== FALSE) {
           $base_url .= '&' . $where_str;
       } else {
           $base_url .= '?' . $where_str;
       }
       //显示分页效果
       $page_string = $this->pagination($base_url, $count, $cur_page, $page_size);
       $this->assign('page_string', $page_string);
       //默认出10条数据
       $list = $this->product->get_list($where, $page_string['offset'], $page_size);
       foreach ($list as &$item) {
           if (empty($item['PRIVATE_ATTR'])) {
               continue;
           }
           $channels = [];
           foreach (json_decode($item['PRIVATE_ATTR'], true) as $name => $url) {
               $channels[] = $name . ': ' . $url;
           }
           $item['PRIVATE_ATTR'] = implode(';', $channels);
       }

       $ids       = array_column($this->product->get_cate_id(), 'CATEGORY_ID');
       $cate_data = $this->pc_category->get_Cate_by_id($ids);
       $this->assign('cate_data', $cate_data);
       $this->assign('data_list', array(
           'list'  => $list,
           'count' => $count
       ));
       $this->display('pc/pc_product/index.html');
   }

   /**
    * 添加产品
    */
   public function add() {
       $this->get_config_list('template');
       $pc_category = $this->pc_category->getCategory();
       $this->assign('pc_category', $pc_category);
       $this->display('pc/pc_product/add.html');
   }

   /**
    * 添加PC产品逻辑

   public function doAdd() {
       $name = $this->_replaceParamVal(trim($this->input->post('name')));
       if (empty($name)) $this->output(EXIT_ERROR, '名称不能为空或含有非法字符！');

       $name_label = $this->_replaceParamVal(trim($this->input->post('name_label')));
       if (!empty($this->input->post('name_label')) && empty($name_label)) {
           $this->output(EXIT_ERROR, '小标签不能含有非法字符！');
       }
       $company = $this->_replaceParamVal(trim($this->input->post('company')));

       $type = intval(trim($this->input->post('type')));
       $type = empty($type) ? 0 : $type - 1;

       $url = $this->_replaceParamVal($this->input->post('url'));
       if (!empty($url) && strlen($url) > 900) {
           $this->output(EXIT_ERROR, 'URL太长！');
       }

       $category_id = $this->_replaceParamVal($this->input->post('category_id'));
       if (empty($category_id)) $this->output(EXIT_ERROR, '分类不能为空！');

       $price_desc = $this->_replaceParamVal(trim($this->input->post('price_desc')));
       if (!empty($this->input->post('price_desc')) && empty($price_desc)) {

       }
       $price_desc_spec = $this->_replaceParamVal(trim($this->input->post('price_desc_spec')));

    */

       $this->db->from($this->table_name)
           ->select('ID, NAME')
           ->where(['STATUS' => 0]);
       if ($where) {
           $this->db->where($where);
       }
       $this->db->where('PARENT_ID IS NULL');
       $this->db->order_by('SORT ASC');
       return $this->db->get()->result_array();

    *
    * @return array
    */
   public function getCategory() {
