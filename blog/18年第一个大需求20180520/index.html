18年第一个大需求

工作 pingan 2018/05/20

主要从数据库的设计、后台设计、以及接口三个方面探讨

1、db的设计

这次的改版主要用于官网的保险页面改版，产品+分组+分类的设计思想，区别在于分组和产品是一对多的关系（用到了关系表），分类和产品是一对一的关系。
CREATE TABLE “PA18NEWDATA”.”PC_CATEGORY” (
“ID” VARCHAR2(16 BYTE) NOT NULL ,
“NAME” VARCHAR2(100 BYTE) DEFAULT ‘’ NOT NULL ,
“PLATE_CODE” VARCHAR2(25 BYTE) DEFAULT ‘’ NULL ,
“CLASS_ICON” VARCHAR2(35 BYTE) DEFAULT ‘’ NULL ,
“TYPE” NUMBER(1) DEFAULT 0 NULL ,
“URL” VARCHAR2(1000 BYTE) NULL ,
“SORT” NUMBER(10) DEFAULT 1 NULL ,
“PARENT_ID” VARCHAR2(16 BYTE) NULL ,
“LOGO” VARCHAR2(255 BYTE) DEFAULT ‘’ NULL ,
“PIC” VARCHAR2(255 BYTE) DEFAULT ‘’ NULL ,
“STATUS” VARCHAR2(4 BYTE) DEFAULT 0 NULL ,
“CREATE_TIME” VARCHAR2(11 BYTE) DEFAULT 0 NULL ,
“UPDATE_TIME” VARCHAR2(11 BYTE) DEFAULT 0 NULL ,
“SEO_TITLE” VARCHAR2(255 BYTE) DEFAULT ‘’ NULL ,
“SEO_KEYWORD” VARCHAR2(255 BYTE) DEFAULT ‘’ NULL ,
“SEO_DESC” VARCHAR2(500 BYTE) DEFAULT ‘’ NULL
)
LOGGING
NOCOMPRESS
NOCACHE

;
COMMENT ON TABLE "PA18NEWDATA"."PC_CATEGORY" IS 'PC分类表';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."ID" IS '表ID';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."NAME" IS '分组名称';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."PLATE_CODE" IS '板块';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."CLASS_ICON" IS '前端类名';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."TYPE" IS '类型：0 文本，1 H5，2 原生';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."URL" IS '链接';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."SORT" IS '排序';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."PARENT_ID" IS '父级';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."LOGO" IS 'logo';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."PIC" IS '分组图片';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."STATUS" IS '状态（0启用，1禁用）';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."CREATE_TIME" IS '添加时间';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."UPDATE_TIME" IS '更新时间';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."SEO_TITLE" IS 'seo标题';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."SEO_KEYWORD" IS 'seo关键词';
COMMENT ON COLUMN "PA18NEWDATA"."PC_CATEGORY"."SEO_DESC" IS 'seo描述';

ALTER TABLE "PA18NEWDATA"."PC_CATEGORY" ADD CHECK ("ID" IS NOT NULL);
ALTER TABLE "PA18NEWDATA"."PC_CATEGORY" ADD CHECK ("NAME" IS NOT NULL);

ALTER TABLE "PA18NEWDATA"."PC_CATEGORY" ADD PRIMARY KEY ("ID");
